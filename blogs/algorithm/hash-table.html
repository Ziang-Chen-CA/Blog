    # Valid Anagram (No.242)

    Given two strings s and t, return true if t is an anagram of s, and false otherwise.

    An Anagram is a word or phrase formed by rearranging the letters of a different word or phrase, typically using all the original letters exactly once.

    My solution:

    class Solution:
    def isAnagram(self, s: str, t: str) -> bool:
        s_dic = Counter(s)
        t_dic = Counter(t)
        return s_dic == t_dic

    # Find Common Characters (No.1002)

    Given a string array words, return an array of all characters that show up in all strings within the words (including duplicates). You may return the answer in any order.

    My solution:

    class Solution:
    def commonChars(self, words: List[str]) -> List[str]:
        common_char = Counter(words[0])
        for each in words[1:]:
            common_char &= Counter(each)
        return list(common_char.elements())

    # Intersection of Two Arrays (No.349)

    Given two integer arrays nums1 and nums2, return an array of their intersection. Each element in the result must be unique and you may return the result in any order.

    My solution is like:

    class Solution:
    def intersection(self, nums1: List[int], nums2: List[int]) -> List[int]:
        dist_num1 = Counter(nums1)
        dist_num2 = Counter(nums2)
        common = dist_num1 & dist_num2
        return list(common)

    class Solution:
    def isHappy(self, n: int) -> bool:
        shown_nums = set()
        shown_nums.add(n)
        while True:
            n = self.happyProcess(n)
            if n == 1:
                return True
            if n in shown_nums:
                return False
            shown_nums.add(n)

        sum = 0

            sum += (n%10) * (n%10)

        return sum

    # Ransom Note (no.383)

    Given two strings ransomNote and magazine, return true if ransomNote can be constructed by using the letters from magazine and false otherwise.

    Each letter in magazine can only be used once in ransomNote.

    class Solution:
    def canConstruct(self, ransomNote: str, magazine: str) -> bool:
        dist_book = Counter(magazine)
        dist_note = Counter(ransomNote)
        if dist_book & dist_note == dist_note:
            return True
        return False

    # Two Sum (No.1)

    Given an array of integers nums and an integer target, return indices of the two numbers such that they add up to target.

    A classic one, my solution:

    class Solution:
    def twoSum(self, nums: List[int], target: int) -> List[int]:
        shown = {}
        for index, value in enumerate(nums):
            rest = target - value
            if rest in shown:
                return [index, nums.index(rest)]
            shown[value] = index

    # 4 SUM II (No.454)

    Given four integer arrays nums1, nums2, nums3, and nums4 all of length n, return the number of tuples (i, j, k, l) such that:

    • 0 <= i, j, k, l < n
    • nums1[i] + nums2[j] + nums3[k] + nums4[l] == 0

    My solution works more on math than code. Make it two pairs and the two sum-up shall be opposite numbers.

    class Solution:
    def fourSumCount(self, nums1: List[int], nums2: List[int], nums3: List[int], nums4: List[int]) -> int:
        res = 0
        combination = {}
        for num1 in nums1:
            for num2 in nums2:
                if num1 + num2 in combination:
                    combination[num1 + num2] += 1
                else:
                    combination[num1 + num2] = 1
        for num3 in nums3:
            for num4 in nums4:
                if -(num3 + num4) in combination:
                    res += combination[-(num3 + num4)]
        return res

    # 3 Sum (No.15)

    Given an integer array nums, return all the triplets [nums[i], nums[j], nums[k]] such that i != j, i != k, and j != k, and nums[i] + nums[j] + nums[k] == 0.

    Notice that the solution set must not contain duplicate triplets.

    My solution:

    class Solution:
    def threeSum(self, nums: List[int]) -> List[List[int]]:
        res = []
        nums.sort()
        for i in range(len(nums)):
            if nums[i] > 0:# not >= for the 0 0 0 case
                break
            if i >= 1 and nums[i] == nums[i-1]:
                continue
            left = i + 1
            right = len(nums) - 1
            while left < right:
                sum_up = nums[left] + nums[right] +nums[i]
                if sum_up > 0:
                    right -= 1
                elif sum_up < 0:
                    left += 1                    
                else:# sum_up == 0:
                    res.append((nums[i], nums[left], nums[right]))
                    left += 1
                    right -= 1
                    while left < right and nums[left] == nums[left - 1]:
                        left += 1 
                    while left < right and nums[right] == nums[right + 1]:
                        right -= 1
        return res

    The idea is that two pointer can reduce time complexity.

    The difficulty lies that duplicates happen and boundary:

    1, after we find a effective one we do shifting the pointer to find the next different value to avoid duplicates.

    2, notice the index. After left go one to right we can use left - 1 index to check and make sure the next one is different, the same as right +1. Also, the i index should be check with i-1 only after i > 0.

    Also, note the while notation as the two-type conditions for PROCESS and QUIT: the intended condition to go on the process (nums[left] == nums[left - 1]) and how you should exit it (left < right) to follow the constrains. If without the latter one, you have to find exactly why the while loop got no boundary problem. In the case, if I don't use left < right, we have to clearly know the 0 0 0 case is the only exception and cope it separately. So, it is simpler to use the notation mentioned.

    # 4 Sum(No.18)

    Given an array nums of n integers, return an array of all the unique quadruplets [nums[a], nums[b], nums[c], nums[d]] such that:

    • 0 <= a, b, c, d < n
    • a, b, c, and d are distinct.
    • nums[a] + nums[b] + nums[c] + nums[d] == target

    You may return the answer in any order.

    Mine as:

    class Solution:
    def fourSum(self, nums: List[int], target: int) -> List[List[int]]:
        res = []
        nums.sort()
        for i in range(len(nums) - 3):
            if i > 0 and nums[i] == nums[i-1]:
                continue
            for j in range(i+1, len(nums) -2):
                if j > i+1 and nums[j] == nums[j - 1]:
                    continue
                left = j + 1
                right = len(nums) - 1
                while left < right:
                    rest_target = target - nums[i] - nums[j]
                    rest_value = nums[left] + nums[right]
                    if rest_value < rest_target:
                        left += 1
                    elif rest_value > rest_target:
                        right -= 1
                    elif rest_value == rest_target:
                        res.append([nums[i], nums[j], nums[left], nums[right]])
                        left += 1
                        right -= 1                         
                        while left < right and nums[left - 1] == nums[left]:
                            left += 1
                        while left < right and nums[right + 1] == nums[right]:
                            right -= 1                                   
        return res
