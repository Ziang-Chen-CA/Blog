    # Implement Queue using Stacks (No.232)

    Implement the MyQueue class:

    • void push(int x) Pushes element x to the back of the queue.
    • int pop() Removes the element from the front of the queue and returns it.
    • int peek() Returns the element at the front of the queue.
    • boolean empty() Returns true if the queue is empty, false otherwise.

    My solution:

    class MyQueue:

    def __init__(self):
        self.stack_i = []
        self.stack_o = []
        

    def push(self, x: int) -> None:
        self.stack_i.append(x)

    def pop(self) -> int:
        #py empty pop would throw an index error
        if self.empty():
            return None
        if self.stack_o:
            return self.stack_o.pop()
        else:
            for i in range(len(self.stack_i)):
                self.stack_o.append(self.stack_i.pop())
            return self.stack_o.pop()      

    def peek(self) -> int:
        if self.empty():
            return None
        if not self.stack_o:
            for i in range(len(self.stack_i)):
                self.stack_o.append(self.stack_i.pop())
        res = self.stack_o.pop()   
        self.stack_o.append(res)
        return res   

    def empty(self) -> bool:
        return not (self.stack_i or self.stack_o)

    # Implement Stack using Queues (No.225)

    Implement a last-in-first-out (LIFO) stack using only two queues. The implemented stack should support all the functions of a normal stack (push, top, pop, and empty).

    My solution:

    class MyStack {
public:
    queue<int> the_queue;
    MyStack() {
        
    }
    
    void push(int x) {
        the_queue.push(x);
    }
    
    int pop() {
        int size = the_queue.size();
        for(int i = 1; i < size; i++){
            the_queue.push(the_queue.front());
            the_queue.pop();
        }
        int res = the_queue.front();
        the_queue.pop();
        return res;
    }
    
    int top() {
        return the_queue.back();
    }
    
    bool empty() {
        return the_queue.empty();
    }
};

    # Valid Parentheses (No.20)

    Given a string s containing just the characters '(', ')', '{', '}', '[' and ']', determine if the input string is valid.

    My answer:

    class Solution {
public:
    bool isValid(string s) {
        stack<char> stk;
        for(char c : s){
            switch(c){
                case '(': 
                case '[': 
                case '{': stk.push(c); break;
                case ')': 
                    if(stk.empty() || stk.top() != '('){
                        return false;} 
                    stk.pop();
                    break;
                case ']': 
                    if(stk.empty() || stk.top() != '['){
                        return false;} 
                    stk.pop();
                    break;
                case '}': 
                    if(stk.empty() || stk.top() != '{'){
                        return false;} 
                    stk.pop();
                    break;
            }
        }
        if (stk.empty()){
            return true;
        }else return false;
    }
};

    # Remove All Adjacent Duplicates In String (NO.1047)

    You are given a string s consisting of lowercase English letters. A duplicate removal consists of choosing two adjacent and equal letters and removing them.

    Input: s = "azxxzy"
Output: "ay"
    1
    2

    My answer:

    class Solution {
public:
    string removeDuplicates(string s) {
        stack<char> stk;
        for(int i = 0; i < size(s); i++){
            if(stk.empty() || s[i] != stk.top()){
                stk.push(s[i]);
            }else{
                stk.pop();
            }
        }
        string res = "";
        while(!stk.empty()){
            res += stk.top();
            stk.pop(); 
        }
        reverse(res.begin(), res.end());
        return res;
    }
};

    # Reverse Polish Notation (No.150)

    You are given an array of strings tokens that represents an arithmetic expression in a Reverse Polish Notation. The division between two integers always truncates toward zero.

    My answer is straight:

    class Solution:
    def evalRPN(self, tokens: List[str]) -> int:
        stack = []
        for i in range(len(tokens)):
            if tokens[i] not in ['+' , '-' , '*' , '/']:
                stack.append(int(tokens[i]))
            elif tokens[i] == '+':
                a = stack.pop()
                b = stack.pop()
                c = a + b
                stack.append(c)
            elif tokens[i] == '-':
                a = stack.pop()
                b = stack.pop()
                c = b - a
                stack.append(c)
            elif tokens[i] == '*':
                a = stack.pop()
                b = stack.pop()
                c = a * b
                stack.append(c)
            elif tokens[i] == '/':
                a = stack.pop()
                b = stack.pop()
                c = int(b / a)
                stack.append(c)
                # note that c = b // a is not right cuz  it is floor division not int() for truncates toward zero
        res = stack.pop()
        return res
